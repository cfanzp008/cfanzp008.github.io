#include <gtest/gtest.h>
#include "demo.cpp"

TEST(add, case1) {
  EXPECT_LT(2, add(0, 2));
  EXPECT_EQ(7, add(5, 2));
  ASSERT_LT(-2, add(1, 2));
  ASSERT_EQ(3, add(1, 2));
}

int main(int argc, char **argv) {
  testing::InitGoogleTest(&argc, argv);
  return RUN_ALL_TESTS();
}
