package main

import "github.com/gin-gonic/gin"

func main() {
    // 创建一个默认的Gin引擎
    r := gin.Default()

    // 定义一个简单的GET请求路由
    r.GET("/", func(c *gin.Context) {
        c.String(200, "Hello, Gin!")
    })

    // 启动HTTP服务器，监听在 8080 端口
    r.Run(":8080")
}
