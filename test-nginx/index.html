events {
    use epoll;
    worker_connections  1024;
}


stream {
    split_clients "$remote_addr$remote_port" $split_ip {
        50% 192.168.2.3;
        *   192.168.2.4;
    }
    upstream tcp_proxy {
        server 192.168.1.100:8000;
    }
    server {
        listen 10000;// 监听8998端口
        proxy_pass tcp_proxy;
        proxy_bind $split_ip;
    }
}
