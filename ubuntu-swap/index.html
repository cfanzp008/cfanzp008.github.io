#创建一个空白文件
sudo dd if=/dev/zero of=/swapfile bs=1M count=2048
#设置文件权限
sudo chmod 600 /swapfile
#使mkswap实用程序将文件格式化为SWAP空间
sudo mkswap /swapfile
#vim /etc/fstab
/swapfile swap swap defaults 0 0

# 查看swap分区
sudo swapon --show
##或者
sudo free -h
