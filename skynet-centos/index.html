unzip protobuf-2.6.1.zip
cd protobuf-2.6.1
tar zxvf protobuf.tar.gz
cp tools/googletest/googletest-release-1.5.0.zip ./protobuf-2.6.1/
cd protobuf-2.6.1
mv googletest-release-1.5.0 gtest
./autogen.sh
./configure
make -jn //多核编译[n为cpu核数]
make check
make install
